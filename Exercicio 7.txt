//Escrever um algoritmo que leia o nome e as três notas obtidas por um aluno de 0 a 10
durante o semestre. Calcular a soma das notas e sua média (aritmética), informar o
nome e sua menção aprovado (media >= 7), Reprovado (media <= 5) e Recuperação
(média entre 5.1 a 6.9).

Algoritmo "Controle_Notas"

Var
   Nome: caractere
   n1, n2, n3, soma, media: real

Inicio
  
   Escreva("Digite o nome do aluno: ")
   Leia(nome)

   Escreva("Digite a primeira nota: ")
   Leia(1)

   Escreva("Digite a segunda nota: ")
   Leia(n2)

   Escreva("Digite a terceira nota: ")
   Leia(n3)

   // Processamento
   soma <- n1 + n2 + n3
   media <- soma / 3

   
   Escreva("Aluno: ", nome)
   Escreva("Soma das notas: ", soma:2:2)
   Escreva("Média Final: ", media:2:2)

   Se (media >= 7) Entao
      Escreva("Menção: APROVADO")
   Senao
      Se (media <= 5) Entao
         Escreva("Menção: REPROVADO")
      Senao
         Escreva("Menção: RECUPERAÇÃO")
     

FimAlgoritmo
